def check_sum_first_i_equals_i(numbers):
    """
    Check if for all valid i, the sum of the first i integers equals i.
    
    We check for i from 1 to len(numbers) (inclusive):
    - sum(first 1 element) == 1
    - sum(first 2 elements) == 2
    - sum(first 3 elements) == 3
    - ...
    - sum(first len(numbers) elements) == len(numbers)
    
    Args:
        numbers: List of integers to check
        
    Returns:
        bool: True if condition holds for all valid i, False otherwise
    """
    if not numbers:  # Empty list - no conditions to check
        return True
    
    # Check for each i from 1 to length of list
    for i in range(1, len(numbers) + 1):
        # Sum of first i elements
        sum_first_i = sum(numbers[:i])
        
        # Check if sum equals i
        if sum_first_i != i:
            return False
    
    return True


def detailed_analysis(numbers):
    """
    Provide detailed step-by-step analysis of the sum conditions.
    
    Args:
        numbers: List of integers to analyze
    """
    print(f"Input: {numbers}")
    print(f"Length: {len(numbers)}")
    print("-" * 50)
    
    if not numbers:
        print("Empty list - condition vacuously true")
        return True
    
    all_valid = True
    
    # Check each position
    for i in range(1, len(numbers) + 1):
        first_i_elements = numbers[:i]
        sum_first_i = sum(first_i_elements)
        is_valid = sum_first_i == i
        
        status = "✓" if is_valid else "✗"
        print(f"i={i}: sum({first_i_elements}) = {sum_first_i}, equals {i}? {status}")
        
        if not is_valid:
            all_valid = False
    
    print("-" * 50)
    print(f"Overall result: {all_valid}")
    print()
    
    return all_valid


def analyze_pattern(numbers):
    """
    Analyze what pattern the numbers follow and why they succeed/fail.
    
    Args:
        numbers: List of integers to analyze
    """
    if not numbers:
        return
    
    print("Pattern Analysis:")
    print("-" * 30)
    
    # Show what each element contributes
    running_sum = 0
    for i, num in enumerate(numbers, 1):
        running_sum += num
        required_sum = i
        difference = running_sum - required_sum
        
        print(f"Position {i}: element={num}, running_sum={running_sum}, "
              f"required={required_sum}, difference={difference}")
    
    # Analyze if there's a pattern in the elements
    if len(numbers) > 1:
        differences = [numbers[i+1] - numbers[i] for i in range(len(numbers)-1)]
        print(f"Element differences: {differences}")
        
        if all(d == differences[0] for d in differences):
            print(f"Elements form arithmetic sequence with difference {differences[0]}")
        else:
            print("Elements don't form a simple arithmetic sequence")
    
    print()


def test_function():
    """
    Test the function with provided examples and additional test cases.
    """
    test_cases = [
        ([0, 1, 2, 3, 4, 5], False),
        ([1, 1, 1, 1, 1, 1], True),
        ([2, 2, 2, 2, 2], False),
        # Additional test cases
        ([1], True),              # sum([1]) = 1 ✓
        ([2], False),             # sum([2]) = 2, but should equal 1
        ([1, 1], True),           # sum([1])=1 ✓, sum([1,1])=2 ✓
        ([1, 0], False),          # sum([1])=1 ✓, sum([1,0])=1 ≠ 2 ✗
        ([0], False),             # sum([0]) = 0 ≠ 1 ✗
        ([], True),               # Empty list - vacuously true
        ([1, 1, 1], True),        # All sums work
        ([3, -1, 0], True),       # sum([3])=3≠1 ✗... wait, let me recalculate
    ]
    
    # Let me recalculate the last test case
    # [3, -1, 0]: sum([3])=3≠1, so this should be False
    test_cases[-1] = ([3, -1, 0], False)
    # Let's add a correct one: [1, 1, 1] gives sum([1])=1✓, sum([1,1])=2✓, sum([1,1,1])=3✓
    
    print("Testing Sum of First i Equals i:")
    print("=" * 60)
    
    for i, (test_list, expected) in enumerate(test_cases, 1):
        print(f"\nTest Case {i}:")
        print("=" * 60)
        
        result = check_sum_first_i_equals_i(test_list)
        status = "✓" if result == expected else "✗"
        
        print(f"Status: {status}")
        detailed_analysis(test_list)
        
        if len(test_list) <= 6:  # Only show pattern analysis for shorter lists
            analyze_pattern(test_list)
        
        print(f"Expected: {expected}, Got: {result}")
        
        if result != expected:
            print("❌ MISMATCH!")
        else:
            print("✅ CORRECT!")
        print()


def find_valid_sequences(max_length=6):
    """
    Find some valid sequences of given lengths.
    
    Args:
        max_length: Maximum length to search for
    """
    print(f"Finding valid sequences up to length {max_length}:")
    print("-" * 50)
    
    for length in range(1, max_length + 1):
        print(f"\nLength {length} sequences:")
        
        # For length n, we need to find a1, a2, ..., an such that:
        # a1 = 1
        # a1 + a2 = 2  =>  a2 = 1
        # a1 + a2 + a3 = 3  =>  a3 = 0
        # a1 + a2 + a3 + a4 = 4  =>  a4 = 1
        # etc.
        
        if length >= 1:
            sequence = [1]  # First element must be 1
            
            for i in range(2, length + 1):
                # Current sum is sum(sequence)
                # We need total sum to be i
                # So next element should be i - sum(sequence)
                next_element = i - sum(sequence)
                sequence.append(next_element)
            
            # Verify this sequence works
            is_valid = check_sum_first_i_equals_i(sequence)
            print(f"  {sequence} - Valid: {is_valid}")
